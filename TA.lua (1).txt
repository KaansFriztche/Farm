-- Gradient fonksiyonun
function gradient(text, startColor, endColor)
	local result = ""
	local length = #text
	for i = 1, length do
		local t = (i - 1) / math.max(length - 1, 1)
		local r = math.floor((startColor.R + (endColor.R - startColor.R) * t) * 255)
		local g = math.floor((startColor.G + (endColor.G - startColor.G) * t) * 255)
		local b = math.floor((startColor.B + (endColor.B - startColor.B) * t) * 255)
		local char = text:sub(i, i)
		result = result .. "<font color=\"rgb(" .. r ..", " .. g .. ", " .. b .. ")\">" .. char .. "</font>"
	end
	return result
end

local folderPath = "WindUI"
makefolder(folderPath)

local function DosyaKaydet(dosyaAdi, veri)
    local dosyaYolu = folderPath .. "/" .. dosyaAdi .. ".json"
    local jsonData = HttpService:JSONEncode(veri)
    writefile(dosyaYolu, jsonData)
end

local function DosyaYukle(dosyaAdi)
    local dosyaYolu = folderPath .. "/" .. dosyaAdi .. ".json"
    if isfile(dosyaYolu) then
        local jsonData = readfile(dosyaYolu)
        return HttpService:JSONDecode(jsonData)
    end
end

local function DosyalariListele()
    local dosyalar = {}
    for _, dosya in ipairs(listfiles(folderPath)) do
        local dosyaAdi = dosya:match("([^/]+)%.json$")
        if dosyaAdi then
            table.insert(dosyalar, dosyaAdi)
        end
    end
    return dosyalar
end

-- Takım bazlı toggle sistemi
local espEnabled = {}
local function createESP(player, teamColor)
	local char = player.Character
	if not char then return end

	-- Box ESP (Highlight)
	if enabledESPStyles.Box and not char:FindFirstChild("ESP_Highlight") then
		local highlight = Instance.new("Highlight")
		highlight.Name = "ESP_Highlight"
		highlight.FillTransparency = 0.75
		highlight.OutlineTransparency = 0
		highlight.FillColor = teamColor
		highlight.OutlineColor = Color3.new(0,0,0)
		highlight.Adornee = char
		highlight.Parent = char
	end

	-- Name ESP (Billboard)
	if enabledESPStyles.Name and not char:FindFirstChild("ESP_Name") then
		local head = char:FindFirstChild("Head")
		if head then
			local billboard = Instance.new("BillboardGui")
			billboard.Name = "ESP_Name"
			billboard.Adornee = head
			billboard.Size = UDim2.new(0, 100, 0, 30)
			billboard.AlwaysOnTop = true
			billboard.StudsOffset = Vector3.new(0, 2, 0)

			local label = Instance.new("TextLabel", billboard)
			label.Size = UDim2.new(1, 0, 1, 0)
			label.BackgroundTransparency = 1
			label.Text = player.Name
			label.TextColor3 = teamColor
			label.TextStrokeTransparency = 0.5
			label.TextScaled = true

			billboard.Parent = head
		end
	end
end

local function removeESP(player)
	local char = player.Character
	if not char then return end
	if char:FindFirstChild("ESP_Highlight") then char.ESP_Highlight:Destroy() end
	if char:FindFirstChild("Head") and char.Head:FindFirstChild("ESP_Name") then
		char.Head.ESP_Name:Destroy()
	end
end

local function updateESPForTeam(team)
	for _, player in ipairs(Players:GetPlayers()) do
		if player.Team == team then
			if espEnabled[team.Name] then
				createESP(player, team.TeamColor.Color)
			else
				removeESP(player)
			end
		end
	end
end

RunService.Heartbeat:Connect(function()
    if not teamNotifEnabled then return end

    for _, player in ipairs(Players:GetPlayers()) do
        local currentTeam = player.Team and player.Team.Name or "None"
        local lastTeam = LastTeams[player.UserId]

        if currentTeam ~= lastTeam then
            LastTeams[player.UserId] = currentTeam
            if player ~= Players.LocalPlayer then
                sendNotification(player.Name .. " takıma degsitirildi '" .. currentTeam .. "'!")
            end
        end
    end
end)
